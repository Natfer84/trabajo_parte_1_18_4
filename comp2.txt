import React, { useState } from 'react';

function TaskForm({ addTask }) {
  const [title, setTitle] = useState('');
  const [description, setDescription] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    const newTask = {
      id: Date.now(),
      title,
      description,
    };
    addTask(newTask);
    setTitle('');
    setDescription('');
  };

  return (
    <form id="task-form" className="formulario" onSubmit={handleSubmit}>
      <input
        type="text"
        id="task-title"
        className="form-control"
        placeholder="Título de la tarea"
        value={title}
        onChange={(e) => setTitle(e.target.value)}
        autoFocus
      />
      <textarea
        id="task-description"
        className="form-control"
        placeholder="Descripción de la tarea"
        cols="10"
        rows="5"
        value={description}
        onChange={(e) => setDescription(e.target.value)}
      ></textarea>
      <input
        type="submit"
        value="Enviar Tarea"
        className="btn btn-lg btn-primary boton"
        id="task-button"
      />
    </form>
  );
}

export default TaskForm;
